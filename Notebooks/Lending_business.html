#KPIs

#Database of issued loans
loan_db = dataframe_db.loc[dataframe_db['approval_flag'] == 1]

#Average loan term
alt = loan_db['term_days'].mean()
#Total outstanding balance
tob = loan_db['total_repayment_expected'].sum() - dataframe_db['total_repayment_received'].sum()
#Expected income
ei = loan_db['total_repayment_expected'].sum() - dataframe_db['requested_amount'].sum()
#Realized income
ri = loan_db['total_repayment_received'].sum() - dataframe_db['requested_amount'].sum()
#Net yield
ny = ri/loan_db['requested_amount'].sum()

print(f'Average loan term is {alt:.0f} days.')
print(f'Total outstanding balance is {tob:,.0f}.')
print(f'Expected income is {ei:,.0f}.')
print(f'Realized income is {ri:,.0f}.')
print(f'Net yield is {ny:,.0f}.')
#Average loan term is a bit over 1 month - short term loan product.
#Total outstanding balance, which represents total unpaid receivables, is over 48M.
#Realized income is negative because total repayments received so far are lower than total principal issued. 
#This typically happens when many loans are still in repayment (not fully repaid yet) and/or some loans pay back only partially / default.

#Age between 30 and 39
adult_req = len(dataframe_db.loc[(dataframe_db['customer_birth_date']>'1985-09-03')&(dataframe_db['customer_birth_date']<='1995-09-03')])
#Age between 18 and 29
young_req = len(dataframe_db.loc[(dataframe_db['customer_birth_date']>'1995-09-03')&(dataframe_db['customer_birth_date']<='2007-09-03')])

#Number of clients by age who received a loan

#Age over 50
senior_appr = len(loan_db.loc[(dataframe_db['customer_birth_date']<='1975-09-03')])
#Age between 40 and 49
middle_appr = len(loan_db.loc[(dataframe_db['customer_birth_date']>'1975-09-03')&(dataframe_db['customer_birth_date']<='1985-09-03')])
#Age between 30 and 39
adult_appr = len(loan_db.loc[(dataframe_db['customer_birth_date']>'1985-09-03')&(dataframe_db['customer_birth_date']<='1995-09-03')])
#Age between 18 and 29
young_appr = len(loan_db.loc[(dataframe_db['customer_birth_date']>'1995-09-03')&(dataframe_db['customer_birth_date']<='2007-09-03')])

#Create a df with ages data
client_ages = pd.DataFrame({
'ages':['18-29','30-39','40-49','50+'],
'loan_requested':[young_req, adult_req, middle_req, senior_req],
'loan_approved':[young_appr, adult_appr, middle_appr, senior_appr]}).set_index('ages')

#Calculate how many loans were not approved
client_ages['loan_not_approved'] = client_ages['loan_requested'] - client_ages['loan_approved']

#Turn into a long df
ages_long = client_ages[['loan_approved', 'loan_not_approved']].reset_index().melt(
    id_vars = 'ages',
    value_vars = ['loan_approved', 'loan_not_approved'],
    var_name = 'status',
    value_name = 'count'
)

#Calculate percentage
ages_long['percent'] = ages_long.groupby('ages')['count'].transform(lambda x: x / x.sum() * 100)

#Create stacked percentage chart
sns.set_theme(style="whitegrid")
plt.figure(figsize=(14, 7))
sns.histplot(data = ages_long,
             x = 'ages',
             weights = 'count',
             hue = 'status',
             hue_order = ['loan_not_approved', 'loan_approved'],
             multiple = 'fill',
             shrink = 0.8)
plt.ylabel('Percentage')
plt.xlabel('Loan Approval Rates by Age Group')
plt.gca().yaxis.set_major_formatter(lambda y, x: f'{y*100:.0f}%')
plt.legend(title = 'Status', labels = ['Approved', 'Not Approved'])

#Debt-to-Income Ratio (DTI)
dataframe_db['dti'] = (dataframe_db['overdue_external_balance'] / dataframe_db['verified_monthly_income'])*100
#Mean DTI
mean_dti_loan_approved = dataframe_db.loc[dataframe_db['approval_flag'] == 1, 'dti'].mean().round(2)
mean_dti_loan_requested = dataframe_db.loc[dataframe_db['approval_flag'] == 0, 'dti'].mean().round(2)
#Median DTI 
median_dti_loan_approved = dataframe_db.loc[dataframe_db['approval_flag'] == 1, 'dti'].median().round(2)
median_dti_loan_requested = dataframe_db.loc[dataframe_db['approval_flag'] == 0, 'dti'].median().round(2)
#Percentiles
percentiles_approved_dti = dataframe_db.loc[dataframe_db['approval_flag'] == 1, 'dti'].quantile([0.25, 0.5, 0.75, 0.9]).round(2) 
percentiles_requested_dti = dataframe_db.loc[dataframe_db['approval_flag'] == 0, 'dti'].quantile([0.25, 0.5, 0.75, 0.9]).round(2) 
#High risk clients
hrc_dti = dataframe_db.loc[(dataframe_db['approval_flag'] == 1) & (dataframe_db['dti'] > 40)]
hrc_dti_percentage = (len(hrc_dti)/len(loan_db))*100

print(f'Average DTI for clients who receive the loan is {mean_dti_loan_approved}, for those who did not it is {mean_dti_loan_requested}')
print(f'Median DTI for clients who receive the loan is {median_dti_loan_approved}, for those who did not it is {median_dti_loan_requested}')
print(f'{hrc_dti_percentage:.0f}% of approved clients can be considered high risk due to DTI being over 40%')
sns.boxplot(x='approval_flag', y='dti', data=dataframe_db, showfliers=False)
plt.xlabel('Loan issued')
plt.ylabel('DTI')
#plt.savefig('dti.png')
#DTI here is overdue_external_balance / verified_monthly_income.
#Median DTI is 0% for both approved and rejected clients, meaning most applicants have no overdue external debt.
#Average DTI is higher for rejected clients (8.76%) than approved clients (5.32%), so higher overdue burden is linked to lower approval.
#Because showfliers=False, the boxplot hides extreme outliers (which is why the chart looks small even though some DTI values are very high).
#4% of approved clients have DTI > 40%, which can be treated as a higher-risk segment.
